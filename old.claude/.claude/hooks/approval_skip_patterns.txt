# Slack approval skip patterns
# Format: <action>:<prefix_pattern>
# action: skip (no approval needed) or require (always require approval)
# Evaluated top-to-bottom; first match wins.
# Note: Shell metachar/redirect detection (>, >>, <, $(...), etc.) runs BEFORE
#       these patterns and always forces require regardless of what's listed below.

# --- Always require approval (dangerous operations) ---
require:rm -rf
require:git push
require:git reset --hard
require:docker rm
require:docker rmi
